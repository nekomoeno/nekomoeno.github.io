#include <iostream>
#include <algorithm>

using namespace std;

const int N = 100010;
int a[N];

int main() {
    int n; cin >> n;
    for(int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    
    sort(a, a + n);
    long long ans = 0;
    for(int i = 0, j = n - 1; i < j; ++i, --j)
        ans += a[j] - a[i];
    
    cout << ans << endl;
    
    
    return 0;
}
